TMStartSession Code Example

long session_handle;
short port = 1;

/* attempt to get a session on Port */
session_handle = TMStartSession(port);

if (session_handle > 0)
{
    /* call TMEX API function with session_handle */
    ...
}
else
{
    /* fail condition, port not available */
    ...
}

/* relinquish port */
TMEndSession(session_handle);
